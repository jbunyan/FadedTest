            attachmentsNativeURL.map(function(attachmentNativeURL) {
                return attachmentNativeURL.replace(/^file:\/\/\/+/, "/");

            params = [
            "com.good.gcs",
            "com.good.gfeservice.send-email",
            "1.0.0.0",
            "sendEmail",
            {
                "to": toRecipients || [],
                "cc": ccRecipients || [],
                "bcc": bccRecipients || [],
                "subject": subject || "",
                "body": body || ""
            },
            attachments
        ];

        cordovaExec(onSuccess, onError, "GDAppKinetics", "callAppKineticsService", params);
    };
